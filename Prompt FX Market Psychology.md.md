
မင်းကို real time 4 hour or 1 day market data ေတွ ပေး မယ် Open price, highest price, lowest price, close price တန်ဖိုးတွေပါမယ် ဆိုရင် မင်းက market ကိုဖတ်ပေးနိုင်မလား

---
ဟုတ်ကဲ့၊ ကျွန်တော် လုပ်ပေးနိုင်ပါတယ်။

ခင်ဗျားအနေနဲ့ အချိန်နဲ့တပြေးညီ (Real-Time) 4-Hour (H4) ဒါမှမဟုတ် Daily (D1) Timeframe အတွက် Open, High, Low, Close (OHLC) တန်ဖိုးတွေ ပေးမယ်ဆိုရင် ကျွန်တော်က အဲဒီ Data ကို အခြေခံပြီး **အောက်ပါအတိုင်း Market ကို ဖတ်ပေး၊ ခွဲခြမ်းစိတ်ဖြာပေးနိုင်ပါတယ်။**

**ကျွန်တော် လုပ်ဆောင်ပေးနိုင်မယ့်အရာတွေ**

၁။ **Candlestick Pattern ဖော်ထုတ်ခြင်း (Pattern Recognition)**
- ပေးလိုက်တဲ့ Candle ရဲ့ Open, High, Low, Close ကို ကြည့်ပြီး Hammer, Doji, Hanging Man, Shooting Star, Engulfing, Spinning Top စတဲ့ Pattern တစ်ခုခုနဲ့ ကိုက်ညီမကိုက်ညီ ချက်ချင်း စစ်ဆေးပေးပါမယ်။

၂။ **Trend နှင့် Momentum သုံးသပ်ခြင်း**
- အရင် Candle တွေရဲ့ Data ကိုပါ တွဲပေးရင် (ဥပမာ - နောက်ဆုံး ၅ ခု၊ ၁၀ ခု) လက်ရှိ Trend က Uptrend လား၊ Downtrend လား၊ Sideways လားဆိုတာ ဖော်ပြပေးပြီး အခု Candle အသစ်ရဲ့ သက်ရောက်မှုကို ပြောပြပါမယ်။

၃။ **Market Psychology (စိတ်ပညာ) ရှင်းပြခြင်း**
- အဲဒီ Candle ဖြစ်ပေါ်လာပုံကို အခြေခံပြီး ဝယ်သူနဲ့ ရောင်းသူကြားက အားပြိုင်မှု၊ ဘယ်သူက အသာစီးရနေတယ်ဆိုတာ ရှင်းပြပါမယ်။

၄။ **Support/Resistance ခွဲခြမ်းစိတ်ဖြာခြင်း**
- ပေးလိုက်တဲ့ High/Low တန်ဖိုးတွေနဲ့ ယခင် Data တွေကို ကြည့်ပြီး အနီးဆုံး Support နဲ့ Resistance Level တွေကို ခန့်မှန်းပေးနိုင်ပါတယ်။

၅။ **Trading အတွက် အကြံပြုချက်များ**
- Entry, Stop Loss, Take Profit အတွက် ဖြစ်နိုင်ခြေရှိတဲ့ Strategic Level တွေကို Risk Management နဲ့အတူ ညွှန်ပြပေးနိုင်ပါတယ်။

**ဘယ်လိုပေးပို့ရမလဲ။**

ဒီလိုမျိုး Data ထည့်ပေးဖို့ အကြံပြုချင်ပါတယ်-

ဒါဆိုရင် အဲဒီ Candle က Hammer လား၊ Doji လား၊ Trend ဘယ်လိုဖြစ်နေပြီလဲ၊ ဘာဆက်ဖြစ်နိုင်လဲ စသဖြင့် အသေးစိတ်တွက်ချက်ပေးပါမယ်။

def calc_body_wicks(open_, high, low, close):
    body = abs(close - open_)
    upper_wick = high - max(open_, close)
    lower_wick = min(open_, close) - low
    total_range = high - low
    body_ratio = body / total_range if total_range > 0 else 0
    return body, upper_wick, lower_wick, total_range, body_ratio

def identify_trend(df, n=5):
    if len(df) < n:
        return "neutral"
    closes = df['Close'].values[-n:]
    highs = df['High'].values[-n:]
    lows = df['Low'].values[-n:]
    # Simple higher highs/higher lows for uptrend
    if all(highs[i] >= highs[i-1] for i in range(1, n)) and all(lows[i] >= lows[i-1] for i in range(1, n)):
        return "uptrend"
    elif all(highs[i] <= highs[i-1] for i in range(1, n)) and all(lows[i] <= lows[i-1] for i in range(1, n)):
        return "downtrend"
    else:
        return "sideways"

def identify_single_candle_pattern(open_, high, low, close, trend, prev_candle=None):
    body, upper_wick, lower_wick, total_range, body_ratio = calc_body_wicks(open_, high, low, close)
    bullish = close > open_
    
    # Doji
    if body_ratio < 0.05:
        if lower_wick > upper_wick * 2:
            return "Dragonfly Doji (ပုစဉ်းရင်ကွဲ)"
        elif upper_wick > lower_wick * 2:
            return "Gravestone Doji (သင်္ချိုင်းကျောက်)"
        else:
            return "Doji (ဒိုဂျီ)"
    
    # Spinning Top
    if 0.05 <= body_ratio <= 0.3 and upper_wick > body and lower_wick > body:
        return "Spinning Top (စပင်နင်ထော့)"
    
    # Hammer / Hanging Man
    if lower_wick >= 2 * body and upper_wick <= 0.3 * body:
        if trend == "downtrend":
            return "Hammer (သံတူ)"
        elif trend == "uptrend":
            return "Hanging Man (ဆွဲကြိုးဆွဲသူ)"
        else:
            return "Hammer/Hanging Man (အတည်ပြုရန်လိုအပ်)"
    
    # Shooting Star / Inverted Hammer
    if upper_wick >= 2 * body and lower_wick <= 0.3 * body:
        if trend == "uptrend":
            return "Shooting Star (ဥက္ကာပျံ)"
        elif trend == "downtrend":
            return "Inverted Hammer (ပြောင်းပြန်သံတူ)"
        else:
            return "Shooting Star/Inverted Hammer (အတည်ပြုရန်လိုအပ်)"
    
    # Marubozu
    if upper_wick < 0.1 * total_range and lower_wick < 0.1 * total_range and body_ratio > 0.7:
        if bullish:
            return "Bullish Marubozu (အားကောင်းတဲ့ အစိမ်းရောင်)"
        else:
            return "Bearish Marubozu (အားကောင်းတဲ့ အနီရောင်)"
    
    # Weak Bullish / Bearish
    if bullish and upper_wick > body and lower_wick < 0.2 * body:
        return "Weak Bullish Candle (အားပျော့တဲ့ Bullish)"
    if not bullish and lower_wick > body and upper_wick < 0.2 * body:
        return "Weak Bearish Candle (အားပျော့တဲ့ Bearish)"
    
    return "သာမန် Candle (ထင်ရှားတဲ့ Pattern မရှိ)"

def check_engulfing(current, prev):
    if prev is None:
        return None
    prev_bullish = prev['Close'] > prev['Open']
    curr_bullish = current['Close'] > current['Open']
    prev_body = abs(prev['Close'] - prev['Open'])
    curr_body = abs(current['Close'] - current['Open'])
    if prev_bullish and not curr_bullish and curr_body > prev_body and current['Open'] > prev['Close'] and current['Close'] < prev['Open']:
        return "Bearish Engulfing (ကျဆင်းဖုံးအုပ်မှု)"
    if not prev_bullish and curr_bullish and curr_body > prev_body and current['Open'] < prev['Close'] and current['Close'] > prev['Open']:
        return "Bullish Engulfing (တက်ကြွဖုံးအုပ်မှု)"
    return None

    if "Engulfing" in final_pattern:
        if "Bullish" in final_pattern:

            print("   - ယခင် Bullish candle ကို Bearish candle က ဖုံးအုပ်သွားသည်။ ဈေးကျရန် အလားအလာများသည်။")
    
    # Common trading advice
    print("\n💡 **Trading Suggestion:**")
    if "Bullish" in final_pattern or "Hammer" in final_pattern or "Dragonfly" in final_pattern or "Inverted Hammer" in final_pattern:
        print("   - Bullish Signal ဖြစ်၍ ဝယ်ယူရန် စဉ်းစားနိုင်ပါသည်။")

# ===========================
def fetch_forex_data(ticker, timeframe='1d', period='7d'):
    """ Yahoo Finance ကနေ Forex Data ဆွဲပါ """
    # timeframe က yahoo အတွက် interval (1d, 1h, 4h မရှိ)
    if timeframe == 'D1':
        interval = '1d'
    elif timeframe == 'H4':
        # Yahoo 4h မရှိပါက 1h ကို resample လုပ်ပါမယ်
        interval = '1h'
    else:
        interval = '1d'
    
    data = yf.download(ticker, period=period, interval=interval, progress=False)
    if data.empty:
        return None
    
    # H4 ဆိုရင် 4H ပြောင်းပါ
    if timeframe == 'H4' and interval == '1h':
        data = data.resample('4H').agg({
            'Open': 'first',
            'High': 'max',
            'Low': 'min',
            'Close': 'last'
        }).dropna()
    
    # Column တွေကို ရိုးရိုးထားပါ
    data = data[['Open', 'High', 'Low', 'Close']]
    return data

def main():
